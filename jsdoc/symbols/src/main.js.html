 94 /**
 95  * Queries db to populate an array of the stops (stop). If stops are detected, calls find_closest() to determine the nearest stop, else alerts the user that no stops were found.
 96  * @param {PositionError object} onError The values retuned from the GeoLocation API
 97  */
 98 function closest_stop(){
 99     var stop = new Array();
100 
101     db.transaction(function(tx){
102         tx.executeSql('SELECT stop_lat, stop_lon, stop_id, stop_name FROM stops;', [], function(tx, result) { 
103             if (result != null && result.rows != null) { 
104                 for (var i = 0; i < result.rows.length; i++) { 
105                     var row = result.rows.item(i); 
106                     stop[i] = new Array(row.stop_lat, row.stop_lon, row.stop_id, row.stop_name); 
107                 }
108                 find_closest(stop);
109 
110             } else{
111                 navigator.notification.alert("No stop located");
112             }
113 
114         }, function(e){
115             console.log("Error: " + e.message);
116         });
117         
118     });
119 }

121 /**
122  * Finds the stop (from stop[]) closest to current location 
123  *   If change = 0 calls update_name() to display the current stop name to get first stop, 
124         else calls display_stops() to display the array of closest stops in order 
125  * @param {Array} stop  Array of the form [ lat, lon, stop_id, stop_name]
126  */
127 function find_closest(stop){
128     
129     var R = 6371; // km
130     
131     //lat1 & long1 = current lat/long
132     //lat2 & long2 = stop coors
133     var dLat; 
134     var dLon; 
135 
136     var count = 0;
137     for(count = 0; count < stop.length; count++){
138         //for each stop coord
139         var tempLat = stop[count][0];
140         var tempLon = stop[count][1];
141 
142         dLat = (lat - tempLat);
143         dLat = dLat*Math.PI/180;
144         dLon = (lon - tempLon);
145         dLon = dLon*Math.PI/180;
146         
147         var rlat1 = lat*Math.PI/180;
148         var rlat2 = stop[count][0]*Math.PI/180;
149         
150         var a = Math.sin(dLat/2) * Math.sin(dLat/2) +
151         Math.sin(dLon/2) * Math.sin(dLon/2) * Math.cos(rlat1) * Math.cos(rlat2); 
152         
153         var c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a)); 
154         
155         var distance = R * c;
156         
157         stop[count][0] = distance;
158     }
159         
160     if(count >= stop.length) {
161         stop.sort(sortMultiDimensionalFirst); 
162         bus_stop = stop[0][2];
163         if(change == 0){
164             update_name(stop[0][3]);
165         } else {
166             display_stops(stop);
167         }
168     }
169 }

259 /**
260  * Sorts the incoming multidimensional array using the fourth element    
261  * @param {Array} a Array within the array to be sorted
262  * @param {Array} b Array within the array to be sorted
263  */
264 function sortMultiDimensional(a,b){
265     return ((a[3] < b[3]) ? -1 : ((a[3] > b[3]) ? 1 : 0));
266 }
267 
268 /**
269  * Sorts the incoming multidimensional array using the third element   
270  * @param {Array} a Array within the array to be sorted
271  * @param {Array} b Array within the array to be sorted
272  */
273 function sortMultiDimensionalH(a,b){
274     return ((a[2] < b[2]) ? -1 : ((a[2] > b[2]) ? 1 : 0));
275 }
276 
277 /**
278  * Sorts the incoming multidimensional array using the first element    
279  * @param {Array} a Array within the array to be sorted
280  * @param {Array} b Array within the array to be sorted
281  */
282 function sortMultiDimensionalFirst(a,b){
283     return ((a[0] < b[0]) ? -1 : ((a[0] > b[0]) ? 1 : 0));
284 }

303 /**
304  * Determines if current day is listed in calendar_dates.txt (from GTFS) 
305     if current day is listed populate service_id [] with relevant service_id's and call find_times() with service_id[]
306     else call find_date()
307  */
308 function find_holidays(){
309     db.transaction(function(tx){
310        // an array of service id's for current day
311         var service_id = new Array();
312 
313         // if holiday, use special service_id
314         tx.executeSql("SELECT service_id FROM calendar_dates WHERE exception_type = 1 AND date = (SELECT strftime('%Y','NOW') || strftime('%m','now') || strftime('%d','now'));", [], function(tx, result) { 
315 
316             if(result != null && result.rows != null && result.rows.length > 0 ){
317                 for (var i = 0; i < result.rows.length; i++) { 
318                     service_id[i] = result.rows.item(i).service_id;
319                 }
320 
321                 find_times(service_id);
322             } else {
323 
324                 find_date();
325             }
326         });
327     });
328 }

330 /**
331  * Determines current day uses db to generate a list of relevant service_id's to populate service_id[]
332     if service_id.length > 0 call find_times()
333     else inform user that no busses returned from query. Log that db may need to be updated
334  */
335 function find_date(){
336     var service_id = new Array();
337     var date = new Date();
338     // current day of the week (0-6)
339     //      0 = sunday, 1 = monday, .. , 6 = saturday
340     var day = date.getDay();
341     var currDay;
342     switch (day)
343     {
344         case 0:
345             currDay = 'sunday';
346             break;
347         case 1:
348             currDay = 'monday';
349             break;
350         case 2:
351             currDay = 'tuesday';
352             break;
353         case 3:
354             currDay = 'wednesday';
355             break;
356         case 4:
357             currDay = 'thursday';
358             break;
359         case 5:
360             currDay = 'friday';
361             break;
362         case 6:
363             currDay = 'saturday';
364             break;
365     }
366     db.transaction(function(tx){ 
367         tx.executeSql("SELECT service_id FROM calendar WHERE start_date <= (SELECT strftime('%Y','NOW') || strftime('%m','now') || strftime('%d','now')) AND end_date >= (SELECT strftime('%Y','NOW') || strftime('%m','now') || strftime('%d','now')) AND " + currDay + " = 1;",[], function(tx, result){
368   
369             if(result != null && result.rows != null){
370                 for (var i = 0; i < result.rows.length; i++) { 
371                     service_id[i] = result.rows.item(i).service_id;
372                 }
373 
374                 find_times(service_id);
375             } else {
376                 conesole.log("No service_id's for today: -- update data!");
377                 alert("No busses today!");
378             }
379         });
380     });
381 }

384 /**
385  * Queries db to get route information for current stop based on current date. Determines the length of time until the scheduled departure time. Sorts the results from lowest duration to highest. Formats data for display_results()
386  * @param {Array} service_id Array within list of relevant service_id's based on current date
387  */
388  
389 function find_times(service_id){
390     var data = new Array();
391     var sched_data = new Array();
392 
393     db.transaction(function(tx){
394         tx.executeSql('SELECT route_short_name, route_long_name, trip_headsign, departure_time, service_id FROM stop_times JOIN trips ON stop_times.trip_id = trips.trip_id JOIN routes ON trips.route_id = routes.route_id WHERE stop_id = "' + bus_stop + '";', [], function(tx, result) { 
395             var date = new Date();
396 
397             // current day of the week (0-6)
398             //      0 = sunday, 1 = monday, .. , 6 = saturday
399             var valid_service_id = 0;
400             if (result != null && result.rows != null) { 
401                 
402                 for (var i = 0; i < result.rows.length; i++) { 
403                     var tuple = result.rows.item(i);
404                     var time = new Array();
405                     valid_service_id = 0;
406                     time = tuple.departure_time.split(":");
407 
408                     for(var j = 0; j < service_id.length; j++){
409                         if(tuple.service_id == service_id[j]){
410                             valid_service_id = 1;
411                         }
412                     }
413                     
414                     if(valid_service_id != 1){
415                         continue;
416                     }
417                     
418                     if(date.getHours() > time[0]){
419                         continue;
420                     } else if (date.getHours() == time[0] && date.getMinutes() > time[1]){
421                         continue;
422                     }
423                     // time_diff[0] = # hours remaining
424                     // time_diff[0] = # mins remaining
425                     var time_diff = new Array(2);
426                     time_diff[0] = time[0] - date.getHours();
427                     time_diff[1] = time[1] - date.getMinutes();
428                     
429                     if(time_diff[1] < 0){
430                         time_diff[0] = time_diff[0] - 1;
431                         time_diff[1] = time_diff[1] + 60;
432                     }
433                     
434                     sched_data[i] = new Array(4);
435                     sched_data[i][0] = tuple.route_short_name;
436                     sched_data[i][1] = tuple.trip_headsign;
437                     sched_data[i][2] = time_diff[0];
438                     sched_data[i][3] = time_diff[1];
439                 
440                 }
441                 
442                 // Sort lists to display shortest wait time first
443                 sched_data.sort(sortMultiDimensional);
444                 sched_data.sort(sortMultiDimensionalH); 
445 
446                 for(var j = 0; j < sched_data.length; j++){
447                     // don't display hours if hours_diff = 0
448                     var str = sched_data[j][0] + " "+ sched_data[j][1] + ": ";
449 
450                     if( sched_data[j][2] == 0){
451                         var str_info = str + sched_data[j][3] + " m";
452 
453                     } else {
454                         var str_info = str + sched_data[j][2] + " h " + sched_data[j][3] + "m";
455                     }
456                     data[j] = str_info.bold();
457 
458                 }
459    
460             } else{
461                 console.log("No busses go here");
462             }
463 
464         }, function(e){
465             console.log("Error: " + e.message);
466         });
467 
468     });
469     display_results(data);
470 }
